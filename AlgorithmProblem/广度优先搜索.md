# 广度优先搜索(BFS)

### leetcode 102  二叉树的层序遍历

```c++
/*BFS解法*/
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        /*先判断当前节点是否为空*/
        if (root == nullptr) {
            return {};
        }
        vector<vector<int>> ans;
        vector<TreeNode*> cur = {root};
        /*遍历处理每一层*/
        while (cur.size()) {
            vector<TreeNode*> nxt;
            vector<int> vals;
            /*遍历当前层的所有节点*/
            for (auto node : cur) {
                vals.push_back(node->val);
                if (node->left) {
                    nxt.push_back(node->left);
                }
                if (node->right) {
                    nxt.push_back(node->right);
                }
            }
            /*存入结果*/
            cur = move(nxt);
            ans.emplace_back(vals);
        }
        return ans;
    }
}
```

### leetcode 103 二叉树的锯齿形层序遍历

```c++
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        /*先判断当前节点是否为空*/
        if (root == nullptr) {
            return {};
        }
        vector<vector<int>> ans;
        vector<TreeNode*> cur = {root};
        /*遍历处理每一层*/
        while (cur.size()) {
            vector<TreeNode*> nxt;
            vector<int> vals;
            /*遍历当前层的所有节点*/
            for (auto node : cur) {
                vals.push_back(node->val);
                if (node->left) {
                    nxt.push_back(node->left);
                }
                if (node->right) {
                    nxt.push_back(node->right);
                }
            }
            /*存入结果*/
            cur = move(nxt);
            /*奇数层反转节点值顺序*/
            if (ans.size() % 2) {
                ranges::reverse(vals);
            }
            ans.emplace_back(vals);
        }
        return ans;
    }
};
```

### leetcode 513 找树左下角的值

```c++
class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        TreeNode* node;
        queue<TreeNode*> q;
        q.push(root);
        /*遍历所有节点*/
        while (!q.empty()) {
            node = q.front();
            q.pop();
            /*先右后左入队*/
            if (node->right) {
                q.push(node->right);
            }
            if (node->left) {
                q.push(node->left);
            }
        }
        /*返回目标节点的值*/
        return node->val;
    }
};
```
