# 堆

### leetcode  295 数据流的中位数

```c++
class MedianFinder {
    /*最大堆，最小堆*/
    priority_queue<int> left;
    priority_queue<int, vector<int>, greater<>> right;

public:
    /*添加数字并动态平衡最大最小两个堆*/
    void addNum(int num) {
        if (left.size() == right.size()) {
            /*两堆大小相等时：添加后需使left堆大小+1*/
            right.push(num);
            left.push(right.top());
            right.pop();
        } else {
            left.push(num);
            right.push(left.top());
            left.pop();
        }
    }
    /*查找当前中位数*/
    double findMedian() {
        if (left.size() > right.size()) {
            return left.top();
        }
        return (left.top() + right.top()) / 2.0;
    }
};
```
