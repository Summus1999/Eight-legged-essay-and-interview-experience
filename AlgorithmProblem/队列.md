# 队列

### leetcode 102 二叉树的层序遍历

```c++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        /*若根节点为空，直接返回空二维数组*/
        if (root == nullptr) {
            return {};
        }
        vector<vector<int>> ans;
        queue<TreeNode*> q;
        q.push(root);
        /*循环处理直到队列为空*/
        while (!q.empty()) {
            vector<int> vals;
            /*保存当前层节点数*/
            for (int n = q.size(); n > 0; n--) {
                auto node = q.front();
                q.pop();
                vals.push_back(node->val);
                /*将非空子节点加入队列*/
                if (node->left)
                    q.push(node->left);
                if (node->right)
                    q.push(node->right);
            }
            /*存入结果*/
            ans.emplace_back(vals);
        }
        return ans;
    }
};
```
