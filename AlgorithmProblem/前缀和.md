## 1.数组+前缀和

### leetcode 238 除自身以外数组的乘积

```c++
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n = nums.size();
        /*计算前缀乘积*/
        vector<int> prefix(n, 1);
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] * nums[i - 1];
        }
        /*计算后缀乘积*/
        vector<int> suffix(n, 1);
        for (int i = n - 2; i >= 0; i--) {
            suffix[i] = suffix[i + 1] * nums[i + 1];
        }
        /*计算前后缀乘积作为答案输出*/
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {
            ans[i] = prefix[i] * suffix[i];
        }
        return ans;
    }
};
```

### leetcode  370  区间加法

```c++
class Solution {
public:
    vector<int> getModifiedArray(int length, vector<vector<int>>& updates) {
        /*初始化差分数组*/
        vector<int> vec(length, 0);
        /*遍历所有更新操作，构造差分数组*/
        for (auto x : updates) {
            vec[x[0]] += x[2];
            if (x[1] + 1 < length) {
                vec[x[1] + 1] -= x[2];
            }
        }
        /*通过前缀和计算最终数组*/
        for (int i = 1; i < length; i++) {
            vec[i] += vec[i - 1];
        }
        return vec;
    }
};

```
